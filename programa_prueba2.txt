PROGRAM

DIM num, digit AS INT

INPUT num

WHILE num > 0 DO
LET digit = num - (10 * (num / 10))

IF digit == 1 THEN
    PRINT "uno "
ELSE IF digit == 2 THEN 
    PRINT "dos "
ELSE IF digit == 3 THEN
    PRINT "tres "
ELSE IF digit == 4 THEN
    PRINT "cuatro "
ELSE IF digit == 5 THEN
    PRINT "cinco "
ELSE IF digit == 6 THEN
    PRINT "seis "
ELSE IF digit == 7 THEN
    PRINT "siete "
ELSE IF digit == 8 THEN
    PRINT "ocho "
ELSE
    PRINT "nueve "
END IF
END IF
END IF
END IF
END IF
END IF
END IF
END IF

LET num = num / 10

LOOP

END
